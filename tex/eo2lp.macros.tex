% custom environment for subfigures
\newenvironment{cmdsubfig}[2]{%
	\begin{subfigure}[t]{\textwidth}
		\subcaption{\label{#1}#2}
		\captionsetup{subrefformat=parens}
		\centering
		\vspace{1mm}
		}{%
		\vspace{1mm}
	\end{subfigure}
}
